\begin{itemize}
    \item Oua, 2 cutii la \textbf{5 lei}
    \item Legume:
        % daca nu sunt rosii la reducere, cumpara 2 kg
        \begin{itemize}
            \item Rosii \textit{3 kg}
            \item Ceapa \textit{2 kg}
            \item Ardei \textit{1 kg}
            \item Morcovi \textit{1 kg}
        \end{itemize}    
    \item Fructe:
        \begin{itemize}
            % daca nu sunt mere verzi, nu mai cumpara
            \item Mere \textit{2 kg}
            \item Banane \textit{4 kg}
            \item Portocale \textit{3 kg}
        \end{itemize}
    \item Carne:
        % daca carnea este mai scumpa de 16 lei\kg, cumpara doar 1 kg din fiecare
        % sotia oricum nu mananca carne
        \begin{itemize} 
            \item Pui \textit{5 kg}
            \begin{itemize}
                \item Piept \textit{2 kg}
                \item Pulpe \textit{3 kg}
            \end{itemize}
            \item Porc \textit{4 kg}
            \item Vita \textit{4 kg}

        \end{itemize}

\end{itemize}
